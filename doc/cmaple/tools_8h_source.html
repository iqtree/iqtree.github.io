1324template <class T1, class T2>
1325void quicksort(T1* arr, int left, int right, T2* arr2 = NULL) {
1326 if (left > right)
1327 return;
1328 assert(left <= right);
1329 int i = left, j = right;
1330 T1 pivot = arr[(left + right) / 2];
1331
1332 /* partition */
1333 while (i <= j) {
1334 while (arr[i] < pivot)
1335 ++i;
1336 while (arr[j] > pivot)
1337 --j;
1338 if (i <= j) {
1339 T1 tmp = arr[i];
1340 arr[i] = arr[j];
1341 arr[j] = tmp;
1342 if (arr2) {
1343 T2 tmp2 = arr2[i];
1344 arr2[i] = arr2[j];
1345 arr2[j] = tmp2;
1346 }
1347 ++i;
1348 --j;
1349 }
1350 };
1351
1352 /* recursion */
1353 if (left < j)
1354 quicksort(arr, left, j, arr2);
1355 if (i < right)
1356 quicksort(arr, i, right, arr2);
1357}
