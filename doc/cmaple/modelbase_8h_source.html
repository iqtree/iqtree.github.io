398template <StateType num_states>
399void cmaple::ModelBase::updateMutationMat() {
400 // update Mutation matrix regarding the pseudo muation count
401 updateMutMatbyMutCount();
402
403 // compute the total rate regarding the root freqs
404 RealNumType total_rate = 0;
405 total_rate -= dotProduct<num_states>(root_freqs, diagonal_mut_mat);
406
407 // record sum as the normalized factor
408 normalized_factor = total_rate;
409
410 // inverse total_rate
411 total_rate = 1.0 / total_rate;
412
413 // normalize the mutation_mat
414 RealNumType* mutation_mat_row = mutation_mat;
415 RealNumType* freqi_freqj_qij_row = freqi_freqj_qij;
416 for (StateType i = 0; i < num_states_; ++i, mutation_mat_row += num_states_,
417 freqi_freqj_qij_row += num_states_) {
418 for (StateType j = 0; j < num_states_; ++j) {
419 mutation_mat_row[j] *= total_rate;
420 // mutation_mat_row[j] /= total_rate;
421
422 // update freqi_freqj_qij
423 if (i != j) {
424 freqi_freqj_qij_row[j] =
425 root_freqs[i] * inverse_root_freqs[j] * mutation_mat_row[j];
426 // root_freqs[i] / root_freqs[j] * mutation_mat_row[j];
427 } else {
428 freqi_freqj_qij_row[j] = mutation_mat_row[j];
429 }
430
431 // update the transposed mutation matrix
432 transposed_mut_mat[row_index[j] + i] = mutation_mat_row[j];
433 }
434
435 // update diagonal
436 diagonal_mut_mat[i] = mutation_mat_row[i];
437 }
438
439 // pre-compute matrix to speedup
440 RealNumType* transposed_mut_mat_row = transposed_mut_mat;
441 RealNumType* freq_j_transposed_ij_row = freq_j_transposed_ij;
442
443 for (StateType i = 0; i < num_states_; ++i,
444 transposed_mut_mat_row += num_states_,
445 freq_j_transposed_ij_row += num_states_) {
446 setVecByProduct<num_states>(freq_j_transposed_ij_row, root_freqs,
447 transposed_mut_mat_row);
448 }
449}

451template <StateType num_states>
452auto cmaple::ModelBase::updateMutationMatEmpiricalTemplate()
453 -> bool {
454 bool update = false;
455
456 if (!fixed_params) {
457 // clone the current mutation matrix
458 RealNumType* tmp_mut_mat = new RealNumType[row_index[num_states_]];
459 memcpy(tmp_mut_mat, mutation_mat,
460 row_index[num_states_] * sizeof(RealNumType));
461
462 // update the mutation matrix regarding the pseu_mutation_count
463 updateMutationMat<num_states>();
464
465 // set update = true if the mutation matrix changes more than a threshold
466 const RealNumType change_thresh = 1e-3;
467 RealNumType sum_change = 0;
468 RealNumType* tmp_mut_mat_ptr = tmp_mut_mat;
469 RealNumType* mutation_mat_ptr = mutation_mat;
470 for (StateType i = 0; i < num_states_;
471 ++i, tmp_mut_mat_ptr += num_states_, mutation_mat_ptr += num_states_) {
472 for (StateType j = 0; j < num_states_; ++j) {
473 sum_change += fabs(tmp_mut_mat_ptr[j] - mutation_mat_ptr[j]);
474 }
475
476 sum_change -= fabs(tmp_mut_mat_ptr[i] - mutation_mat_ptr[i]);
477 }
478
479 update = sum_change > change_thresh;
480
481 // delete tmp_diagonal_mutation_mat
482 delete[] tmp_mut_mat;
483 }
484
485 // return update
486 return update;
487}
